---
title: "Create Zoom OAuth Connection"
api: "POST /api/v1/zoom_oauth_connections"
description: "After being created, the Zoom OAuth connection will be used to generate tokens for the user."
---

## Authentication

Requires API key authentication via the `Authorization` header.

## Request Body

<ParamField body="authorization_code" type="string" required>
  The authorization code received from Zoom during the OAuth flow
</ParamField>

<ParamField body="redirect_uri" type="string" required>
  The redirect URI used to obtain the authorization code
</ParamField>

<ParamField body="zoom_oauth_app_id" type="string">
  The Zoom OAuth App the connection is for
</ParamField>

<ParamField body="is_local_recording_token_supported" type="boolean" default="true">
  Whether the Zoom OAuth Connection supports generating local recording tokens
</ParamField>

<ParamField body="is_onbehalf_token_supported" type="boolean" default="false">
  Whether the Zoom OAuth Connection supports generating onbehalf tokens
</ParamField>

<ParamField body="metadata" type="object">
  JSON object containing metadata to associate with the Zoom OAuth Connection
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the Zoom OAuth connection
</ResponseField>

<ResponseField name="zoom_oauth_app_id" type="string">
  ID of the Zoom OAuth app this connection belongs to
</ResponseField>

<ResponseField name="user_id" type="string">
  Zoom user ID
</ResponseField>

<ResponseField name="account_id" type="string">
  Zoom account ID
</ResponseField>

<ResponseField name="state" type="string">
  Connection state (`connected` or `disconnected`)
</ResponseField>

<ResponseField name="metadata" type="object">
  Metadata associated with the Zoom OAuth connection
</ResponseField>

<ResponseField name="is_local_recording_token_supported" type="boolean">
  Whether the connection supports generating local recording tokens
</ResponseField>

<ResponseField name="is_onbehalf_token_supported" type="boolean">
  Whether the connection supports generating onbehalf tokens
</ResponseField>

<ResponseField name="connection_failure_data" type="object">
  Information about connection failures, if any
</ResponseField>

<ResponseField name="created_at" type="string">
  Timestamp when the connection was created (ISO 8601 format)
</ResponseField>

<ResponseField name="updated_at" type="string">
  Timestamp when the connection was last updated (ISO 8601 format)
</ResponseField>

<ResponseField name="last_successful_sync_at" type="string">
  Timestamp of the last successful sync (ISO 8601 format)
</ResponseField>

<ResponseField name="last_attempted_sync_at" type="string">
  Timestamp of the last attempted sync (ISO 8601 format)
</ResponseField>
