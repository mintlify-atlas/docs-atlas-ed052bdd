---
title: "List Zoom OAuth Connections"
api: "GET /api/v1/zoom_oauth_connections"
description: "Returns a list of Zoom OAuth connections for the authenticated project. Results are paginated using cursor pagination."
---

## Authentication

Requires API key authentication via the `Authorization` header.

## Query Parameters

<ParamField query="cursor" type="string">
  Cursor for pagination
</ParamField>

## Response

<ResponseField name="next" type="string">
  URL for the next page of results (null if no more results)
</ResponseField>

<ResponseField name="previous" type="string">
  URL for the previous page of results (null if on first page)
</ResponseField>

<ResponseField name="results" type="array">
  Array of Zoom OAuth connection objects
  
  <ResponseField name="id" type="string">
    Unique identifier for the Zoom OAuth connection
  </ResponseField>
  
  <ResponseField name="zoom_oauth_app_id" type="string">
    ID of the Zoom OAuth app this connection belongs to
  </ResponseField>
  
  <ResponseField name="user_id" type="string">
    Zoom user ID
  </ResponseField>
  
  <ResponseField name="account_id" type="string">
    Zoom account ID
  </ResponseField>
  
  <ResponseField name="state" type="string">
    Connection state (`connected` or `disconnected`)
  </ResponseField>
  
  <ResponseField name="metadata" type="object">
    Metadata associated with the Zoom OAuth connection
  </ResponseField>
  
  <ResponseField name="is_local_recording_token_supported" type="boolean">
    Whether the connection supports generating local recording tokens
  </ResponseField>
  
  <ResponseField name="is_onbehalf_token_supported" type="boolean">
    Whether the connection supports generating onbehalf tokens
  </ResponseField>
  
  <ResponseField name="connection_failure_data" type="object">
    Information about connection failures, if any
  </ResponseField>
  
  <ResponseField name="created_at" type="string">
    Timestamp when the connection was created (ISO 8601 format)
  </ResponseField>
  
  <ResponseField name="updated_at" type="string">
    Timestamp when the connection was last updated (ISO 8601 format)
  </ResponseField>
  
  <ResponseField name="last_successful_sync_at" type="string">
    Timestamp of the last successful sync (ISO 8601 format)
  </ResponseField>
  
  <ResponseField name="last_attempted_sync_at" type="string">
    Timestamp of the last attempted sync (ISO 8601 format)
  </ResponseField>
</ResponseField>
